CREATE TABLE complaints (
  complaint_id      SERIAL PRIMARY KEY,
  complaint_ref     VARCHAR(12) UNIQUE NOT NULL
                    GENERATED ALWAYS AS ('C-' || LPAD(complaint_id::TEXT,4,'0')) STORED,
  student_id        INT REFERENCES students(student_id) ON DELETE SET NULL,

  submitted_at      TIMESTAMPTZ DEFAULT NOW(),
  resolved_at       TIMESTAMPTZ,
  CONSTRAINT chk_resolved CHECK (
    resolved_at IS NULL OR resolved_at >= submitted_at
  )
);

-- Trigger: auto-log every status change
CREATE OR REPLACE FUNCTION fn_log_status_change()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF OLD.status IS DISTINCT FROM NEW.status THEN
    INSERT INTO audit_log (complaint_id, old_status, new_status)
    VALUES (NEW.complaint_id, OLD.status, NEW.status);
  END IF;
  RETURN NEW;
END;
$$;

👁️ View: vw_complaint_summary
Joined view for admin dashboard — complaint + student + AI data
-- ----------------------------------------------------------------
-- VIEW: vw_complaint_summary
-- ----------------------------------------------------------------
CREATE OR REPLACE VIEW vw_complaint_summary AS
SELECT
  c.complaint_ref,
  c.title,
  c.category,
  c.status,
  c.priority,
  c.is_anonymous,
  c.submitted_at,
  c.resolved_at,
  EXTRACT(EPOCH FROM (COALESCE(c.resolved_at, NOW()) - c.submitted_at))/3600 AS hours_open,
  CASE WHEN c.is_anonymous THEN 'Anonymous'
       ELSE s.first_name || ' ' || s.last_name
  END AS student_name,
  s.reg_number,
  s.faculty,
  d.name AS department,
  sf.full_name AS assigned_to,
  ai.sentiment,
  ai.urgency_score,
  ai.dropout_risk_flag,
  ai.category_confidence
FROM complaints c
LEFT JOIN students   s   ON c.student_id        = s.student_id
LEFT JOIN departments d  ON c.dept_id            = d.dept_id
LEFT JOIN staff       sf ON c.assigned_staff_id  = sf.staff_id
LEFT JOIN ai_analysis ai ON c.complaint_id       = ai.complaint_id;
👁️ View: vw_dropout_risk
Early-warning students ranked by AI dropout risk score
-- ----------------------------------------------------------------
-- VIEW: vw_dropout_risk
-- ----------------------------------------------------------------
CREATE OR REPLACE VIEW vw_dropout_risk AS
SELECT
  s.student_id,
  s.reg_number,
  s.first_name || ' ' || s.last_name AS full_name,
  s.faculty,
  s.year_of_study,
  s.dropout_risk_score,
  COUNT(c.complaint_id)                     AS total_complaints,
  COUNT(c.complaint_id) FILTER (WHERE c.status = 'pending')   AS pending_count,
  COUNT(c.complaint_id) FILTER (WHERE c.status = 'escalated') AS escalated_count,
  ROUND(AVG(ai.urgency_score)*100, 1)       AS avg_urgency_pct,
  MAX(c.submitted_at)                       AS last_complaint_date,
  CASE
    WHEN s.dropout_risk_score >= 75 THEN 'CRITICAL'
    WHEN s.dropout_risk_score >= 50 THEN 'HIGH'
    WHEN s.dropout_risk_score >= 25 THEN 'MEDIUM'
    ELSE 'LOW'
  END AS risk_level
FROM students s
LEFT JOIN complaints   c   ON s.student_id   = c.student_id
LEFT JOIN ai_analysis  ai  ON c.complaint_id = ai.complaint_id
GROUP BY s.student_id, s.reg_number, s.first_name, s.last_name,
         s.faculty, s.year_of_study, s.dropout_risk_score
ORDER BY s.dropout_risk_score DESC;
